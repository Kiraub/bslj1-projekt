218  public void DrawLine(GraphCoord graphStart, GraphCoord graphEnd, Color fillColor, bool is_redraw = false)
219  {
220  if (!is_redraw && RememberDrawing)
221  {
222  LineMemory.Add(new SimpleLine(graphStart, graphEnd, fillColor));
223  }
224  ImageCoord start = Graph2Image(graphStart);
225  ImageCoord end = Graph2Image(graphEnd);
226  // left and right points rather than start and end
227  ImageCoord Left = start.X < end.X ? start : end;
228  ImageCoord Right = start.X >= end.X ? start : end;
229  // xdiff is always positive or zero
230  float xdiff = Right.X - Left.X;
231  if (xdiff == 0f)
232  {
233  DrawVerticalLine(Left.X, Left.Y, Right.Y, fillColor);
234  }
235  // ydiff is positive if up-slope; negative if down-slope; zero if even line
236  float ydiff = Right.Y - Left.Y;
237  if (ydiff == 0f)
238  {
239  DrawHorizontalLine(Left.Y, Left.X, Right.X, fillColor);
240  }
241  // steepness of the slope
242  float m = ydiff / xdiff;
243  m = (float)Math.Truncate(m * 10000) / 10000;
244  // buffer to even out difference of pixel coord and graph coord
245  float buffer = 0;
246  // remember y to count it up/down through x loops
247  int y = Left.Y;
248  for (int x = Left.X; x < Right.X; x += 1)
249  {
250  // remember vertical distance
251  buffer += m;
252  if (Math.Abs(buffer) > 1)
253  {
254  // draw vertical pixels
255  for (int yc = 1; yc < Math.Abs(buffer); yc += 1)
256  {
257  this.SetPixel(new ImageCoord(x, y + (yc * Math.Sign(m))), fillColor);
258  }
259  y += (int)Math.Truncate(buffer);
260  buffer = (float)Math.Truncate((buffer % 1.0) * 10000) / 10000;
261  }
262  else
263  {
264  // draw horizontal pixel
265  this.SetPixel(new ImageCoord(x, y), fillColor);
266  }
267  }
