212  public void DrawLine(GraphCoord graphStart, GraphCoord graphEnd, Color fillColor, bool is_redraw = false)
213  {
214  if (!is_redraw && RememberDrawing)
215  {
216  LineMemory.Add(new SimpleLine(graphStart, graphEnd, fillColor));
217  }
218  ImageCoord start = Graph2Image(graphStart);
219  ImageCoord end = Graph2Image(graphEnd);
220  // left and right points rather than start and end
221  ImageCoord Left = start.X < end.X ? start : end;
222  ImageCoord Right = start.X >= end.X ? start : end;
223  //System.Diagnostics.Debug.Print("Drawing Graph-Coords\n\t" + graphStart.ToString() + "\n\t" + graphEnd.ToString());
224  //System.Diagnostics.Debug.Print("Eq Pixel-Coords\n\t" + Left.ToString() + "\n\t" + Right.ToString());
225  // xdiff is always positive or zero
226  float xdiff = Right.X - Left.X;
227  if (xdiff == 0f)
228  {
229  DrawVerticalLine(Left.X, Left.Y, Right.Y);
230  }
231  // ydiff is positive if up-slope; negative if down-slope; zero if even line
232  float ydiff = Right.Y - Left.Y;
233  if (ydiff == 0f)
234  {
235  DrawHorizontalLine(Left.Y, Left.X, Right.X);
236  }
237  // steepness of the slope
238  float m = ydiff / xdiff;
239  m = (float)Math.Truncate(m * 10000) / 10000;
240  // buffer to even out difference of pixel coord and graph coord
241  float buffer = 0;
242  // remember y to count it up/down through x loops
243  int y = Left.Y;
244  for (int x = Left.X; x < Right.X; x += 1)
245  {
246  // remember vertical distance
247  buffer += m;
248  if (Math.Abs(buffer) > 1)
249  {
250  // draw vertical pixels
251  for (int yc = 1; yc < Math.Abs(buffer); yc += 1)
252  {
253  this.SetPixel(new ImageCoord(x, y + (yc * Math.Sign(m))), fillColor);
254  }
255  y += (int)Math.Truncate(buffer);
256  buffer = (float)Math.Truncate((buffer % 1.0) * 10000) / 10000;
257  }
258  else
259  {
260  // draw horizontal pixel
261  this.SetPixel(new ImageCoord(x, y), fillColor);
262  }
263  }
