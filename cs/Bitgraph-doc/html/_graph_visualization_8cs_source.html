92  public void DrawLine(ImageCoord start, ImageCoord end, Color fillColor)
93  {
94  // left and right points rather than start and end
95  ImageCoord Left = start.X < end.X ? start : end;
96  ImageCoord Right = start.X >= end.X ? start : end;
97  // xdiff is always positive or zero
98  float xdiff = Right.X - Left.X;
99  if (xdiff == 0f)
100  {
101  DrawVerticalLine(Left.X, Left.Y, Right.Y, fillColor);
102  }
103  // ydiff is positive if up-slope; negative if down-slope; zero if even line
104  float ydiff = Right.Y - Left.Y;
105  if (ydiff == 0f)
106  {
107  DrawHorizontalLine(Left.Y, Left.X, Right.X, fillColor);
108  }
109  // steepness of the slope
110  float m = ydiff / xdiff;
111  m = (float)Math.Truncate(m * 10000) / 10000;
112  // buffer to even out difference of pixel coord and graph coord
113  float buffer = 0;
114  // remember y to count it up/down through x loops
115  int y = Left.Y;
116  for (int x = Left.X; x < Right.X; x += 1)
117  {
118  // remember vertical distance
119  buffer += m;
120  if (Math.Abs(buffer) > 1)
121  {
122  // draw vertical pixels
123  for (int yc = 1; yc < Math.Abs(buffer); yc += 1)
124  {
125  this.SetPixel(new ImageCoord(x, y + (yc * Math.Sign(m))), fillColor);
126  }
127  y += (int)Math.Truncate(buffer);
128  buffer = (float)Math.Truncate((buffer % 1.0) * 10000) / 10000;
129  }
130  else
131  {
132  // draw horizontal pixel
133  this.SetPixel(new ImageCoord(x, y), fillColor);
134  }
135  }
136  }
